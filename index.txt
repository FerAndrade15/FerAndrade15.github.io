  <script type="module">
    async function loadProjects() {
      const resp = await fetch("data/projects.json");
      const projects = await resp.json();
      window.PROJECTS = projects;
      document.getElementById("projectsGrid").innerHTML = projects.map(cardHTML).join("");
    }

    function cardHTML(proj) {
      return `
        <div class="col-md-4">
          <div class="card project-card h-100 shadow-sm" data-topic="${proj.topic}" data-year="${proj.year}" data-id="${proj.id}">
            <div class="card-body">
              <h5 class="card-title">${proj.title}</h5>
              <p class="card-text text-truncate">${proj.summary}</p>
              <div class="d-flex justify-content-between align-items-center">
                <small class="text-muted">${proj.year} • ${proj.topic}</small>
                <button class="btn btn-sm btn-outline-primary" onclick="openProject(event,'${proj.id}')">Open</button>
              </div>
            </div>
          </div>
        </div>`;
    }

    function applyFilters() {
      const q = document.getElementById("searchInput").value.toLowerCase();
      const yr = document.getElementById("yearSelect").value;
      document.querySelectorAll("#projectsGrid .col-md-4").forEach(col => {
        const card = col.querySelector(".card");
        const topic = card.dataset.topic.toLowerCase();
        const year = card.dataset.year;
        const title = card.querySelector(".card-title").textContent.toLowerCase();
        const match = (yr === "all" || yr === year) && (q === "" || topic.includes(q) || title.includes(q));
        col.style.display = match ? "" : "none";
      });
    }

    async function openProject(event, id) {
      event.stopPropagation();
      const proj = window.PROJECTS.find(p => p.id === id);
      if (!proj) return;
      document.getElementById("toolArea").style.display = "block";
      const content = document.getElementById("toolContent");
      content.innerHTML = `
        <h5>${proj.title} <small class="text-muted">(${proj.year})</small></h5>
        <p>${proj.summary}</p>
        <div id="interactiveArea"></div>
      `;
      try {
        await pyscript.runtime.runPythonAsync(`load_tool_from_js("${id}")`);
      } catch (e) {
        console.error("Python bridge error:", e);
      }
    }

    document.getElementById("searchInput").addEventListener("input", applyFilters);
    document.getElementById("yearSelect").addEventListener("change", applyFilters);
    loadProjects();
  </script>
